disp("Question 1")
K = [100 0 320;
     0 100 240;
     0 0 1];
R = eye(3);
twp = [0 0 0]';
twpp = [1 0 0]';

Pp = K*[R' -R'*twpp];

disp(Pp)

X2 = [1 0 1 1]';
